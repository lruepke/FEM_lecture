    "kappa   = 1e-6          # Thermal diffusivity of rock [m2/s] \n",

    "dt      = 1*day         # Timestep [s]"

    "nt      = 500                                          # Number of timesteps to compute\n",
    "Xvec,dx = np.linspace(-L/2, L/2, nx,  retstep=True)    # X coordinate vector, constant spacing\n",
    "beta    = kappa*dt/dx**2"
   ]
  },

    "T[0,np.nonzero(np.abs(Xvec) <= W/2)] = Tmagma   # and hot where the dike is\n",
    "time    = 0                                     # track the run time"
   ]
  },
  {
   "cell_type": "markdown",
   "metadata": {},
   "source": [
    "Make a function that computes temperature"
   ]
  },

   "cell_type": "code",

   "outputs": [],
   "source": [
    "def fdm_solve(T):\n",
    "      \n",
    "    for n in range(0, nt-1, 1):\n",
    "        for i in range(1,len(Xvec)-1):\n",
    "            T[n + 1, i] = beta * (T[n][i+1] - 2*T[n][i] + T[n][i-1]) + T[n][i]\n",
    "\n",
    "    return T"
   ]
